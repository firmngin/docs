## AsyncClient

```typescript
import { AsyncClient, ClientConfig, Entity, Event, Payment } from 'firmngin'

async function onPayment(payment: Payment) {
  if (payment.isSuccess) {
    console.log('paid:', payment.orderId)
  }
}

async function main() {
  const config = await ClientConfig.fromFile('keys.json')
  const client = new AsyncClient(config)

  client.on(Event.PAYMENT, onPayment)
  await client.connect()
  await client.pushEntity(new Entity('relay'), 'on')
  await client.run()
}

main().catch(console.error)
```

## Client

```typescript
import { Client, ClientConfig, Entity, EntityCommand } from 'firmngin'

const relay = new Entity('relay')
const temperature = new Entity('temperature')

async function main() {
  const client = new Client(await ClientConfig.fromFile('keys.json'))

  client.onEntity(relay, (command: EntityCommand) => {
    console.log('relay:', command.value)
  })

  await client.connect()
  await client.pushEntity(relay, 'off')
  await client.pushEntity(temperature, 27.5)
  await client.run()
}

main().catch(console.error)
```


```typescript

client.onEntity(new Entity('relay'), (command: EntityCommand) => {

})
```
