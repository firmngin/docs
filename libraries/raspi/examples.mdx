## Basic async client

```python
import asyncio
from firmngin import ClientConfig, Entity, FirmnginClient, KeysConfig

relay = Entity(1)
temperature = Entity("temperature")

async def main():
    config = ClientConfig(keys=KeysConfig.from_file("keys.json"))
    async with FirmnginClient(config) as client:
        await client.connect()
        await client.push_entity(relay, "on")
        await client.push_entity(temperature, 27.5)
        await client.run()

asyncio.run(main())
```

## Command handlers

```python
from firmngin import EntityCommand

@client.on_entity(relay)
async def handle_relay(command: EntityCommand) -> None:
    if command.value == "on":
        GPIO.output(RELAY_PIN, GPIO.HIGH)
    elif command.value == "off":
        GPIO.output(RELAY_PIN, GPIO.LOW)
```

## Sync adapter for blocking scripts

```python
from firmngin import SyncFirmnginClient

client = SyncFirmnginClient(config)
client.connect()
client.push_entity(relay, "off")
client.disconnect()
```
